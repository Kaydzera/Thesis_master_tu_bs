% Declare options for custom and poster
\ifboolexpr{%
    test {\ifdefstring{\tubs@specialformat}{poster}}
    test {\ifdefstring{\tubs@paperformat}{portrait}}
    }{%
        % rescale from 1/7th to 1/11th as specified in toolbox
        \renewcommand{\senderspaceheight}{(\senderspaceheight / 7 ) * 11}%
        \renewcommand{\communicationspaceheight}{(\communicationspaceheight / 7 ) * 11}%
    }{}%
%
\ifboolexpr{%
    test {\ifdefstring{\tubs@specialformat}{poster}}
    test {\ifdefstring{\tubs@paperformat}{landscape}}
    }{%
        % rescale from 1/5th to 1/11th as specified in toolbox
        \renewcommand{\senderspaceheight}{(\senderspaceheight / 5 ) * 11}%
        \renewcommand{\communicationspaceheight}{(\communicationspaceheight / 5 ) * 11}%
    }{}%
%
\ifboolexpr{%
    test {\ifdefstring{\tubs@specialformat}{custom}}
    test {\ifdefstring{\tubs@paperformat}{portrait}}
    }{%
        \renewcommand{\senderspaceheight}{\layoutheight / 7}%
        \renewcommand{\borderwidth}{\senderspaceheight / 5}% approximated by other values
        \renewcommand{\communicationspaceheight}{\layoutheight - \senderspaceheight - \borderwidth}% approximated by remaining mm
    }{}%
%
\ifboolexpr{%
    test {\ifdefstring{\tubs@specialformat}{custom}}
