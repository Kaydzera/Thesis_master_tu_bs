\section{Motivation \& Problem}
% ===================================================================================

\begin{frame}[shrink=5]{Motivation}
  \begin{columns}[T]
    \begin{column}{0.57\textwidth}
      \textbf{Szenario: Produktionsbetrieb}
      \begin{itemize}
        \item Manager entscheidet, \emph{welche AuftrΟΛge angenommen werden} (BudgetbeschrΟΛnkung)
        \item Planer weist AuftrΟΛge Maschinen zu βÄî \emph{minimiert} die Fertigstellungszeit
        \item Manager benΟΕtigt eine Worst-Case-Garantie fΟΦr den Makespan
      \end{itemize}
      Dies ist ein \textbf{zweistufiges Optimierungsproblem}:
      \begin{itemize}
        \item \textbf{Leader} (Manager): wΟΛhlt AuftrΟΛge zur \emph{Maximierung} des Makespans
        \item \textbf{Follower} (Planer): \emph{minimiert} den Makespan
      \end{itemize}
    \end{column}
    \begin{column}{0.40\textwidth}
      \begin{block}{Unser Problem}
        \textbf{Obere Ebene (Leader)}\\
        UnbeschrΟΛnktes Knapsack-Problem\\[0.25em]
        \centering$\downarrow$ Auftragsportfolio\\[0.25em]
        \raggedright\textbf{Untere Ebene (Follower)}\\
        Paralleles Maschinenplanungsproblem\\
        ($P\,||\Cmax$)
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=5]{Zweistufige Optimierung: Definition}
  \begin{block}{Zweistufiges Optimierungsproblem}
    \vspace{-0.3em}
    \begin{align*}
      \max_{x}\quad & F\bigl(x,\,y^*(x)\bigr) & &x\in X\\
      \text{wobei}\quad & y^*(x)\in\arg\min_{y}\,\{f(x,y):y\in Y(x)\}
    \end{align*}
    \vspace{-0.4em}
  \end{block}
  \begin{itemize}
    \item \textbf{Leader} wΟΛhlt $x$ unter \emph{Antizipation} der optimalen Reaktion des Followers
    \item \textbf{Follower} beobachtet $x$ und minimiert sein eigenes Ziel
    \item \textbf{Optimistisch}: bei Indifferenz des Followers wird das fΟΦr den Leader gΟΦnstigste Ergebnis gewΟΛhlt
  \end{itemize}
\end{frame}

\begin{frame}[shrink=22]{Zweistufige Optimierung: Schwierigkeit}
  \begin{alertblock}{RechenkomplexitΟΛt}
    Zweistufige Optimierung ist im Allgemeinen NP-schwer [Frangioni~1995].
  \end{alertblock}
  \begin{block}{Unsere spezifische Herausforderung}
    \begin{itemize}
      \item \textbf{Leader} lΟΕst ein unbeschrΟΛnktes Knapsack-Problem (NP-schwer)
      \item \textbf{Follower} lΟΕst paralleles Maschinenplanungsproblem $P||\Cmax$ (NP-schwer)
      \item Wertfunktion $\Copt(x)$ ist \emph{nicht-konvex} und \emph{unstetig}
      Kein geschlossener Ausdruck fΟΦr $\Copt(x)$: jede Auswertung erfordert das LΟΕsen des Follower-Problems, was kombinatorische Verfahren (z.\,B.\ Branch-and-Bound) auf der oberen Ebene notwendig macht
    \end{itemize}
  \end{block}
  \begin{block}{Konsequenz}
    Allgemeine zweistufige LΟΕser (z.\,B.\ MibS) kΟΕnnen nur sehr kleine Instanzen verarbeiten.
    Wir benΟΕtigen einen \textbf{problemspezifischen exakten Algorithmus}.
  \end{block}
\end{frame}

\begin{frame}[shrink=36]{Das Knapsack-Scheduling-Problem}
  \textbf{Problemdaten:}
  \begin{itemize}
    \item $n$ Auftragstypen, $m$ identische Parallelmaschinen, Budget $B>0$
    \item Auftragstyp $i$: Bearbeitungszeit $d_i>0$, Kosten $p_i>0$; oBdA $d_1\geq\cdots\geq d_n$
    \item \textbf{Leader} wΟΛhlt $x_i\in\mathbb{Z}_+$ Kopien von Typ $i$;
          \textbf{Follower} weist $y_{ik}\in\mathbb{Z}_+$ AuftrΟΛge von Typ $i$ Maschine $k$ zu
  \end{itemize}
  \begin{block}{VollstΟΛndige zweistufige Formulierung}
    \vspace{-0.2em}
    \begin{align*}
      \max_{x\in\mathbb{Z}_+^n}\quad & \Copt(x)\\
      \text{s.t.}\quad & \textstyle\sum_{i=1}^{n}p_ix_i\leq B\\[0.4em]
      \text{wobei}\quad\Copt(x)\;=\;& \min_{y_{ik}\in\mathbb{Z}_+}\;\max_{k=1,\ldots,m}\;\sum_{i=1}^{n}d_i y_{ik}\\
      \text{s.t.}\quad & \textstyle\sum_{k=1}^{m}y_{ik}=x_i\quad\forall\,i \in\{1,\ldots,n\}
    \end{align*}
    \vspace{-0.3em}
  \end{block}
  \smallskip
  Leader \textbf{maximiert} $\Copt(x)$; Follower \textbf{minimiert} es βÄî eine adversarielle zweistufige Struktur.
\end{frame}

\begin{frame}{Warum Enumeration Scheitert}
  Naiver Ansatz: alle zulΟΛssigen $x$ aufzΟΛhlen und fΟΦr jedes das Follower-Problem lΟΕsen.
  \[
    \text{Suchraum} \leq \prod_{i=1}^{n}\!\left(\!\left\lfloor\tfrac{B}{p_i}\right\rfloor+1\right)
  \]
  \begin{block}{Beispiel: $n=10$, $B=100$, $p_i=5{+}i$}
    Obere Schranke: $\mathbf{5{,}29\times 10^{10}}$ Kandidaten\\[0.2em]
    Bei 1\,ms pro Follower-LΟΕsung: $\approx\mathbf{1{,}68\text{ Jahre}}$
  \end{block}
  \bigskip
  \textbf{LΟΕsung:} Branch-and-Bound mit engen oberen Schranken zur aggressiven Baumbereinigung.
\end{frame}

\section{Branch-and-Bound-Algorithmus}
% ===================================================================================

\begin{frame}[shrink=8]{Branch-and-Bound: Ο€berblick}
  \begin{block}{Knotenstruktur auf Tiefe $d$}
    \begin{itemize}
      \item \textbf{Fixierte Entscheidungen:} $x_1,\ldots,x_d$\quad
            $\Brem = B - \sum_{i=1}^{d}p_ix_i$
      \item \textbf{Maschinenlasten:} inkrementeller LPT-Plan fΟΦr Elemente $1,\ldots,d$
      \item \textbf{Kinder:} $x_{d+1}=0,1,\ldots,\lfloor\Brem/p_{d+1}\rfloor$
    \end{itemize}
  \end{block}
  \begin{columns}[T]
    \begin{column}{0.47\textwidth}
      \textbf{Drei Komponenten:}
      \begin{enumerate}
        \item \textbf{Branching} βÄî fixiere $x_{d}$
        \item \textbf{Bounding} βÄî berechne $\UB(d,\Brem)$
        \item \textbf{Pruning} βÄî verwerfe wenn $\UB\leq\Cinc$
      \end{enumerate}
    \end{column}
    \begin{column}{0.50\textwidth}
      \textbf{Designentscheidungen:}
      \begin{itemize}
        \item Tiefensuche $\to$ schnelle Kandidatenaktualisierung $\to$ aggressives Beschneiden
        \item Inkrementelles LPT: $O(m)$ pro Knoten
        \item DP vorberechnet: $O(1)$ Schrankenabfragen
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=18]{Die LPT-Regel βÄî Heuristik des Followers}
  \begin{columns}[T]
    \begin{column}{0.53\textwidth}
      \textbf{Longest Processing Time (LPT)} [Graham~1969]:
      \begin{itemize}
        \item AuftrΟΛge nach nicht-aufsteigender Dauer sortieren
        \item Jeden Auftrag der \emph{am wenigsten belasteten} Maschine zuweisen
      \end{itemize}
      \textbf{Garantie:}\quad $\Copt(x)\leq\Clpt(x)\leq\tfrac{4}{3}\Copt(x)$
      \begin{alertblock}{Inkrementelle Aktualisierungen}
        Verzweigung $d\to d{+}1$: $x_{d+1}$ AuftrΟΛge der Dauer $d_{d+1}$ anhΟΛngen.
        Bestehenden sortierten Maschinenbelastungszustand wiederverwenden βÄî \textbf{keine vollstΟΛndige Neuberechnung}
      \end{alertblock}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{LPT-Algorithmus}
        \footnotesize
        $L_1=\cdots=L_m\leftarrow 0$\\[0.15em]
        \textbf{fΟΦr} jeden Auftrag $j$ (nicht-aufsteigende Reihenfolge):\\
        \quad $k^*\leftarrow\arg\min_k L_k$\\
        \quad $L_{k^*}\leftarrow L_{k^*}+d_j$\\[0.15em]
        \textbf{return} $\Clpt=\max_k L_k$\\[0.2em]
        \normalsize Zeit: $O(J\log m)$
      \end{block}
      \textbf{Praxis:} LPT ist in $\approx84\%$ unserer Testinstanzen \emph{optimal}.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=10]{Beschneiden durch Obere Schranken}
  Auf Tiefe $d$ mit Budget $\Brem$ berechnen wir eine \emph{obere Schranke}:
  \[\Copt(x_1,\ldots,x_n)\;\leq\;\UB(x_1,\ldots,x_d,\Brem)
    \quad\forall\;x_{d+1},\ldots,x_n\text{ (ErgΟΛnzungen)}\]
  \textbf{Beschneiden} wenn $\;\UB\leq\Cinc$.
  \[
    \UB = \underbrace{C_{\max}^{\mathrm{LPT},(d)}}_{\text{fixiert, 1..}d}
         +\;\max_{\sum_{i>d}p_iy_i\leq\Brem}
           \underbrace{\Delta(y_{d+1},\ldots,y_n)}_{\text{Mehraufwand Restelemente}}
  \]
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Ceiling-Schranke}
        $\Delta = \sum_{i>d}d_i\lceil y_i/m\rceil$\\
        Einzelne DP-Tabelle, $O(1)$ Abfrage
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Max-LPT-Schranke}
        $\Delta = \Clpt(y_{d+1},\ldots,y_n)$\\
        DP-Tabelle pro Tiefe, engere Schranke
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Die Ceiling-Schranke}
% ===================================================================================

\begin{frame}[shrink=15]{Ceiling-Schranke: Formulierung}
  \begin{block}{Knapsack-Scheduling-Relaxierung}
    \vspace{-0.3em}
    \[\UB = C_{\max}^{\mathrm{LPT},(d)}
      \;+\;
      \max_{y\geq 0}\;\sum_{i=d+1}^{n}d_i\!\left\lceil\frac{y_i}{m}\right\rceil
      \quad\text{s.t.}\;\sum_{i>d}p_iy_i\leq\Brem\]
    \vspace{-0.3em}
  \end{block}
  \textbf{Intuition:} $\lceil y_i/m\rceil$ AuftrΟΛge vom Typ $i$ auf $m$ gleichwertigen Maschinen erhΟΕhen den Makespan um
  $\lceil y_i/m\rceil\cdot d_i$ βÄî stets eine \emph{Ο€berschΟΛtzung}.
  \begin{block}{DP-Implementierung βÄî $O(1)$ Abfrage pro Knoten}
    \small Reduktion auf unbeschrΟΛnktes Knapsack-Problem. FΟΦr Elementtyp $i$:
    \begin{itemize}\small
      \item \textbf{Einzel:} Kosten $p_i$, Wert $d_i$ (fΟΦr $1,\ldots,m{-}1$ Kopien)
      \item \textbf{Block:} Kosten $m\cdot p_i$, Wert $d_i$ (fΟΦr eine Charge von $m$)
    \end{itemize}
    Vorberechnung in umgekehrter Elementreihenfolge $\Rightarrow$ jede Tiefe-$d$-Abfrage ist ein direkter Tabellenlesezugriff.
  \end{block}
\end{frame}

\begin{frame}[shrink=18]{Ceiling-Schranke: Korrektheitsbeweis}
  \begin{block}{Satz}
    FΟΦr jede zulΟΛssige VervollstΟΛndigung gilt: $\Copt(x_1,\ldots,x_n)\leq\UB$
  \end{block}
  \textbf{Beweisschema:}
  \begin{enumerate}
    \item $\Copt(x)\leq\Clpt(x)$ (LPT ΟΦberschΟΛtzt)
    \item Nach fixierten Elementen ist der aktuelle Makespan $= C_{\max}^{\mathrm{LPT},(d)}$
    \item \emph{Annahme:} alle Maschinen bis $C_{\max}^{\mathrm{LPT},(d)}$ gefΟΦllt βÄî nur Ο€berschΟΛtzung
    \item Auf $m$ gleichmΟΛΟüig belasteten Maschinen erhΟΕht das HinzufΟΦgen von $y_i$ AuftrΟΛgen vom Typ $i$ den Makespan um
          $\lceil y_i/m\rceil\cdot d_i$
    \item Daher $\Clpt(x)\leq C_{\max}^{\mathrm{LPT},(d)}+\sum_{i>d}\lceil x_i/m\rceil\cdot d_i$
    \item Maximierung der rechten Seite ΟΦber $\Brem$ ergibt $\UB\geq\Copt(x)$ \hfill$\square$
  \end{enumerate}
\end{frame}

\section{Die Max-LPT-Schranke}
% ===================================================================================

\begin{frame}[shrink=8]{Das Max-LPT-Problem}
  \begin{block}{Definition}
    Finde $x\in\mathbb{Z}_+^n$ mit $\sum_i p_ix_i\leq B$, das $\Clpt(x)$ \textbf{maximiert}.
  \end{block}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \textbf{Zwei Rollen:}
      \begin{enumerate}
        \item Einmalig gelΟΕst $\Rightarrow$ \textbf{starker anfΟΛnglicher Kandidat} (mit $\tfrac34$-Approximationsgarantie)
        \item FΟΦr Restelemente an jedem Knoten gelΟΕst $\Rightarrow$ \textbf{engere Schranke} als Ceiling
      \end{enumerate}
      BerΟΦcksichtigt die tatsΟΛchliche LPT-Struktur (nicht nur $\lceil\cdot/m\rceil$).
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{alertblock}{Hauptergebnis}
        Max-LPT \textbf{optimal} in $O(n\cdot B)$ per DP gelΟΕst.
      \end{alertblock}
      \begin{block}{Kompromiss}
        \small $n$ tiefenspezifische DP-Tabellen vs.\ eine Ceiling-Tabelle; amortisiert durch besseres Beschneiden.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=15]{Approximationsgarantie: Bausteine}
  \begin{block}{Lemma (Eigenschaften der $\alpha$-Approximation)}
    \vspace{-0.2em}
    \small Sei \texttt{approx} eine $\alpha$-Approximation fΟΦr Minimierung ($\alpha\geq 1$), setze $\beta:=1/\alpha$. Dann gilt fΟΦr alle $x$:
    \[\beta\cdot\mathrm{Opt}(x)\;\leq\;\beta\cdot\mathrm{approx}(x)\;\leq\;\mathrm{Opt}(x)\;\leq\;\mathrm{approx}(x)\]
    \vspace{-0.3em}
  \end{block}
  \begin{block}{Satz (Zweistufige Approx.\ [Weninger \& Fukasawa 2025])}
    \vspace{-0.2em}
    \small Wenn $\beta\cdot\mathrm{Opt}(x)\leq\beta\cdot\mathrm{approx}(x)\leq\mathrm{Opt}(x)\leq\mathrm{approx}(x)$ fΟΦr alle $x$ und $\hat x\in\arg\max_x\,\mathrm{approx}(x)$, dann:
    \[\beta\cdot\mathrm{Opt}_{\text{zweistufig}}\;\leq\;\mathrm{Opt}(\hat x)\;\leq\;\mathrm{Opt}_{\text{zweistufig}}\]
    \vspace{-0.3em}
  \end{block}
  \textbf{Unser Setting:} LPT ist eine $\tfrac{4}{3}$-Approximation fΟΦr $P||\Cmax$ $\Rightarrow$ $\beta=\tfrac{3}{4}$. Max-LPT maximiert $\Clpt$ $\Rightarrow$ Satz direkt anwenden.
\end{frame}

\begin{frame}[shrink=14]{Die $\frac{3}{4}$-Approximationsgarantie}
  \begin{block}{Satz}
    Sei $\hat x$ die Max-LPT-LΟΕsung und $x^*$ die zweistufig-optimale LΟΕsung. Dann gilt:
    \[\tfrac{3}{4}\,\Copt(x^*)\;\leq\;\Copt(\hat x)\;\leq\;\Copt(x^*)\]
  \end{block}
  \textbf{Beweis} (unter Verwendung der zwei Bausteine):
  \begin{enumerate}\small
    \item \textbf{Lemma} ($\alpha=\tfrac{4}{3}$, $\beta=\tfrac{3}{4}$): fΟΦr alle $x$,\quad $\tfrac34\,\Copt(x)\leq\tfrac34\,\Clpt(x)\leq\Copt(x)\leq\Clpt(x)$
    \item $\hat x$ maximiert $\Clpt$, also $\;\Clpt(x^*)\leq\Clpt(\hat x)$
    \item \textbf{Satz [WF25]:}\quad $\tfrac{3}{4}\cdot\mathrm{Opt}_{\text{zweistufig}}\leq\Copt(\hat x)\leq\mathrm{Opt}_{\text{zweistufig}}$
    \item \textbf{VollstΟΛndige Kette:}\quad $\tfrac34\,\Copt(x^*)\leq\tfrac34\,\Clpt(x^*)\leq\tfrac34\,\Clpt(\hat x)\leq\Copt(\hat x)\leq\Copt(x^*)$ \hfill$\square$
  \end{enumerate}
  \begin{alertblock}{Praktische Bedeutung}
    Der Max-LPT-Kandidat ist \textbf{garantiert $\leq 25\%$ suboptimal}.
    In der Praxis ist die LΟΦcke praktisch immer null (LPT ist oft optimal).
  \end{alertblock}
\end{frame}

\begin{frame}[shrink=45]{Max-LPT-Algorithmus: Budget pro Maschine \& Eigenschaften}
  \begin{block}{Strukturelle Beobachtung}
    In jedem LPT-Plan wurde der \textbf{zuletzt geplante Auftrag} $I$ auf die \textbf{am wenigsten belastete Maschine} zum Planungszeitpunkt gelegt:
    \[\Clpt = C_{\max}^{(m)} + d_I\]
    Der Makespan wird durch die \emph{minimale Maschinenlast} unmittelbar vor $I$ plus seiner Dauer $d_I$ bestimmt.
  \end{block}
  FΟΦr einen festen Kandidaten-Letzten-Auftrag $J$ wollen wir $C_{\max}^{(m)}$ fΟΦr Elemente $1,\ldots,J$ maximieren.
  \begin{block}{Zwei Haupteigenschaften des optimalen vorhergehenden Plans}
    \begin{itemize}
      \item \textbf{Eigenschaft 1} (hΟΕchstes Budget pro Maschine):\\
            $B^* := \lfloor B/m \rfloor$ ist das grΟΕΟüte Budget, das gleichmΟΛΟüig jeder Maschine zugewiesen werden kann.\\
            Keine Umverteilung kann der am wenigsten belasteten Maschine mehr als $B^*$ geben.
      \item \textbf{Eigenschaft 2} (beste Elemente fΟΦr $B^*$):\\
            FΟΦr Budget $B^*$ ist die optimale Elementkombination aus Typen $\leq J$ das Optimum des unbeschrΟΛnkten Knapsack-Problems.
    \end{itemize}
    Wenn beide gelten, erzielt kein Plan ein grΟΕΟüeres $C_{\max}^{(m)}$: kaufe $m$ Kopien jedes DP-optimalen Elements und fΟΦhre LPT aus.
  \end{block}
  \begin{block}{Budgetanpassung wenn $\hat{B} := B - mB^* < p_J$}
    \small Rest $\hat{B}$ ist zu klein, um eine Kopie von $J$ zu bezahlen.
    Reduziere $B^*$ um 1 (gleichmΟΛΟüig ΟΦber Maschinen) bis $\hat{B} \geq p_J$.
    Eigenschaft 1 gilt fΟΦr das reduzierte $B^*$; frage $\mathrm{DP}(J,B^*)$ erneut ab.
    Wenn $B^*$ vor $\hat{B} \geq p_J$ null erreicht, ΟΦberspringe $J$ (unzulΟΛssig).
  \end{block}
\end{frame}

\begin{frame}[shrink=45]{Max-LPT-Algorithmus: Korrektheit \& KomplexitΟΛt}
  \begin{block}{Algorithmus (Pseudocode)}
    \small
    DP-Tabelle vorberechnen (unbeschrΟΛnktes Knapsack-Problem, Elemente $1\ldots n$, Budgets $0\ldots B$).\\[0.2em]
    $z^*\leftarrow 0$\hfill\emph{// bisher bester Kandidat}\\
    \textbf{fΟΦr} jeden Typ $J\in\{1,\ldots,n\}$:\\
    \qquad $B^*\leftarrow\lfloor B/m\rfloor$\\
    \qquad \textbf{solange} $(B-mB^*)<p_J$ \textbf{und} $B^*>0$:\; $B^*\leftarrow B^*-1$\\
    \qquad \textbf{wenn} $(B-mB^*)<p_J$: $J$ ΟΦberspringen\hfill\emph{// unzulΟΛssig}\\
    \qquad $z_J\leftarrow\mathrm{DP}(J,B^*)+d_J$;\; $z^*\leftarrow\max(z^*,z_J)$\\
    \textbf{return} $z^*$
  \end{block}
  \begin{columns}[T]
    \begin{column}{0.53\textwidth}
      \begin{block}{Korrektheit}
        \small
        Jede optimale LΟΕsung hat einen letzten Auftrag $I$ (Fall 1 oder 2).\\[0.2em]
        FΟΦr dieses $I$ identifizieren Eigenschaften 1 \& 2 den optimalen vorhergehenden Plan via $\mathrm{DP}(I,B^*)$.\\[0.2em]
        $\max_J z_J$ ΟΦber alle Kandidaten deckt das Optimum ab.\hfill$\square$
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{alertblock}{KomplexitΟΛt: $O(n\cdot B)$}
        \small
        DP-Tabelle in $O(n\cdot B)$ vorberechnet.\\[0.2em]
        Pro Kandidat $J$: $O(1)$ Abfrage + $O(B^*/m)$ Anpassung.\\[0.2em]
        $n$ Kandidaten $\Rightarrow O(n\cdot B)$ gesamt.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{FrΟΦhzeitige OptimalitΟΛtserkennung}
% ===================================================================================

\begin{frame}[shrink=5]{FrΟΦhzeitige OptimalitΟΛtserkennung}
  \begin{block}{SchlΟΦsselungleichung}
    FΟΦr alle $x$: $\Copt(x)\leq\Clpt(x)$ (LPT ΟΦberschΟΛtzt nur)
  \end{block}
  Sei $\hat x\in\arg\max_x\Clpt(x)$. Wenn $\Clpt(\hat x)=\Copt(\hat x)$, dann gilt fΟΦr \emph{alle} zulΟΛssigen $x$:
  \[\Copt(\hat x)=\Clpt(\hat x)\geq\Clpt(x)\geq\Copt(x)\]
  $\Rightarrow$ $\hat x$ ist das \textbf{globale zweistufige Optimum} βÄî kein B\&B-Baum erforderlich!
  \begin{alertblock}{Auswirkung}
    \begin{itemize}
      \item $\Copt(\hat x)$ wird ohnehin berechnet, um den Kandidaten zu initialisieren
      \item Diese Ο€berprΟΦfung kostet \textbf{nichts extra}
      \item In $\approx\mathbf{84{,}6\%}$ unserer Testinstanzen ausgelΟΕst $\Rightarrow$ an der Wurzel gelΟΕst
    \end{itemize}
  \end{alertblock}
\end{frame}

\begin{frame}[shrink=5]{VollstΟΛndiger Algorithmus}
  \begin{columns}[T]
    \begin{column}{0.53\textwidth}
      \small
      \begin{enumerate}
        \item Max-LPT lΟΕsen $\to\hat x$;\quad $\Clpt(\hat x)$, $\Copt(\hat x)$ berechnen
        \item \textbf{Wenn} $\Clpt(\hat x)=\Copt(\hat x)$: \textbf{return} $\hat x$
        \item $\Cinc\leftarrow\Copt(\hat x)$;\quad Wurzel in Stapel $Q$ legen
        \item \textbf{Solange} $Q\neq\emptyset$:
        \begin{itemize}\small
          \item Knoten $(d,x_1,\ldots,x_d,\Brem)$ entnehmen
          \item Wenn $d=n$: Follower-Problem lΟΕsen; $\Cinc$ aktualisieren
          \item Sonst: $\UB(d,\Brem)$ berechnen
          \item Wenn $\UB\leq\Cinc$: \textbf{beschneiden}
          \item Sonst: Kinder einschieben (Tiefensuche)
        \end{itemize}
        \item \textbf{return} incumbent
      \end{enumerate}
      \smallskip
      {\footnotesize Inspiriert durch [Weninger \& Fukasawa 2025].}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Blattoptimierung}
        \small Auf Tiefe $n{-}1$: nur das Kind, das das \emph{maximale} verbleibende Budget ausgibt, ist relevant.
        Alle anderen erzeugen ein kleineres oder gleiches $\Copt$.
      \end{block}
      \begin{block}{Follower-LΟΕser}
        \small Gurobi 12.0.3 [Gurobi Optimization 2024] lΟΕst die MIP-Formulierung von $P||\Cmax$ an BlΟΛttern.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
